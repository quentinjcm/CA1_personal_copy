33 {
34  /* the initial syntax checked for are operation names followed by a string of characters
35  * the parser then goes through the input file line by line, and if any of these meet
36  * the requirements of any of the defined rules, then that line is passed to the corresponding
37  * function that will read the line and perform the required operations
38  */
39  //initial rules that define what function each line is passed to
40  srule comment = spt::comment_p("#");
41  srule sphere = ("SPHERE:" >> *(spt::anychar_p))
42  [bind(&SceneParser::parseSphere, boost::ref(*this), _1)];
43  srule plane = ("PLANE:" >> *(spt::anychar_p))
44  [bind(&SceneParser::parsePlane, boost::ref(*this), _1)];

46  [bind(&SceneParser::parseObj, boost::ref(*this), _1)];

50  [bind(&SceneParser::parseLight, boost::ref(*this), _1)];
51  srule translate = ("TRANSLATE:" >> *(spt::anychar_p))
52  [bind(&SceneParser::parseTranslate, boost::ref(*this), _1)];
53  srule rotate = ("ROTATE:" >> *(spt::anychar_p))
54  [bind(&SceneParser::parseRotate, boost::ref(*this), _1)];
55 
56  // opens the specified file
57  std::ifstream fileIn(m_fileName.c_str());
58

64  spt::parse(line.c_str(), comment | sphere | plane | obj | mat | light | rotate | translate, spt::space_p);

136  if (objMesh.isTriangular()){

148  for (ngl::Face f: faces){

156  ngl::Vec3 v2(verts[f.m_vert[2]]);
157
